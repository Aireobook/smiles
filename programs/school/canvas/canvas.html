  <canvas id="circle" width="150" height="150"></canvas>

  <script>

      let x = 50;
      let y = 40;
      let radius = 20;

      let canvas = document.getElementById("circle");
      let ctx = canvas.getContext("2d");
      ctx.arc(x, y, radius, 0, Math.PI * 2);
      ctx.fillStyle = "red";
      ctx.fill();
  </script>
